Example of the Wait Dialog

JavaScript Code

    $("#showWaitDialogButton").on("click", function () {
        $.when(kendo.ui.ExtWaitDialog.show({ title: "Wait", message: "Waiting for 5 seconds..." })).done(function () {
            setTimeout(function () {
                kendo.ui.ExtWaitDialog.hide();
            }, 5000);
        });
    });

Example of the Input Dialog

JavaScript Code

    $.when(kendo.ui.ExtInputDialog.show({ title: "Input", message: "Please enter your name:", required: true })).done(function (response) {
        if (response.button == "OK") {
            $("#inputDialogUserResponse").append(kendo.format("<div>{0} button clicked at: {1} with input of: {2}</div>", response.button, new Date().toLocaleTimeString(), response.input));
        } else {
            $("#inputDialogUserResponse").append(kendo.format("<div>{0} button clicked at: {1}</div>", response.button, new Date().toLocaleTimeString()));
        }
    });
