Title:
HTML Code

    <input id="givenname" />
    <input id="surname" />
    <div>
        <span>Title:</span> <input id="title" />
    </div>
JavaScript Code

    $("#givenname").kendoExtTextBox({
        placeholder: {
            text: "Enter given name"
        },
        text: "John",
        wrapper: {
            cssClass: "textbox-wrapper"
        }
    });

    $("#surname").kendoExtTextBox({
        placeholder: {
            text: "Enter surname",
            cssClass: "required-field"
        },
        wrapper: {
            cssClass: "textbox-wrapper"
        }
    });

    $("#title").kendoExtTextBox();
CSS Code

    .textbox-wrapper {
        margin-bottom: 5px;
        display:block !important;
    }
    .required-field {
        color: #f78181 !important;
    }
